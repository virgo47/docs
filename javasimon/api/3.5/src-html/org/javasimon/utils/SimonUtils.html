020    /**
021     * SimonUtils provides static utility methods.
022     * <p/>
023     * <h3>Human readable outputs</h3>
024     * Both {@link org.javasimon.Stopwatch} and {@link org.javasimon.Counter} provide human readable
025     * {@code toString} outputs. All nanosecond values are converted into few valid digits with
026     * proper unit (ns, us, ms, s) - this is done via method {@link #presentNanoTime(long)}.
027     * Max/min counter values are checked for undefined state (max/min long value is converted
028     * to string "undef") - via method {@link #presentMinMaxCount(long)}.
029     * <p/>
030     * <h3>Aggregation utilities</h3>
031     * It is possible to sum up (aggregate) values for a subtree for a particular Simon type using
032     * {@link #calculateCounterAggregate(org.javasimon.Simon)} or {@link #calculateStopwatchAggregate(org.javasimon.Simon)}.
033     * Methods come also in versions allowing to filter by {@link org.javasimon.SimonFilter}.
034     * <p/>
035     * <h3>Simon tree operations</h3>
036     * For various debug purposes there is a method that creates string displaying the whole Simon sub-tree.
037     * Here is example code that initializes two random Simons and prints the whole Simon hierarchy
038     * (note that the method can be used to obtain any sub-tree of the hierarchy):
039     * <pre>
040     * Split split = SimonManager.getStopwatch("com.my.other.stopwatch").start();
041     * SimonManager.getCounter("com.my.counter").setState(SimonState.DISABLED, false);
042     * split.stop();
043     * System.out.println(SimonUtils.simonTreeString(SimonManager.getRootSimon()));</pre>
044     * And the output is:
045     * <pre>
046     * (+): Unknown Simon:  [ ENABLED]
047     *   com(+): Unknown Simon:  [com INHERIT]
048     *     my(+): Unknown Simon:  [com.my INHERIT]
049     *       other(+): Unknown Simon:  [com.my.other INHERIT]
050     *         stopwatch(+): Simon Stopwatch: total 24.2 ms, counter 1, max 24.2 ms, min 24.2 ms, mean 24.2 ms [com.my.other.stopwatch INHERIT]
051     *       counter(-): Simon Counter: counter=0, max=undef, min=undef [com.my.counter DISABLED]</pre>
052     * Notice +/- signs in parenthesis that displays effective Simon state (enabled/disabled), further
053     * details are printed via each Simon's {@code toString} method.
054     *
055     * <h3>Other utilities</h3>
056     * It is possible to obtain "local name" of the Simon (behind the last dot) via {@link #localName(String)}
057     * or check if the name is valid Simon name via {@link #checkName(String)}.
058     *
059     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
060     * @author Radovan Sninsky
061     * @since 1.0
062     */

126            /**
127             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
128             * except for seconds that are the biggest unit used. In case of undefined value ({@link Long#MAX_VALUE} or
129             * {@link Long#MIN_VALUE}) string {@link #UNDEF_STRING} is returned, so it can be used for min/max values as well.
130             *
131             * @param nanos time in nanoseconds
132             * @return human readable time string
133             */
134            public static String presentNanoTime(long nanos) {
135                    if (nanos == Long.MAX_VALUE || nanos == Long.MIN_VALUE) {
136                            return UNDEF_STRING;
137                    }
138                    return presentNanoTimePrivate((double) nanos);
139            }
140    
141            /**
142             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
143             * except for seconds that are the biggest unit used.
144             *
145             * @param nanos time in nanoseconds
146             * @return human readable time string
147             */
148            public static String presentNanoTime(double nanos) {
149                    if (nanos == Double.MAX_VALUE) {
150                            return UNDEF_STRING;
151                    }
152                    return presentNanoTimePrivate(nanos);
153            }
154    
155            private static String presentNanoTimePrivate(double time) {
156                    if (Math.abs(time) < 1d) {
157                            return "0";
158                    }
159    
160                    if (time < UNIT_PREFIX_FACTOR) {
161                            return ((long) time) + " ns";
162                    }
163    
164                    time /= UNIT_PREFIX_FACTOR;
165                    if (time < UNIT_PREFIX_FACTOR) {
166                            return formatTime(time, " us");
167                    }
168    
169                    time /= UNIT_PREFIX_FACTOR;
170                    if (time < UNIT_PREFIX_FACTOR) {
171                            return formatTime(time, " ms");
172                    }
173    
174                    time /= UNIT_PREFIX_FACTOR;
175                    return formatTime(time, " s");
176            }
177    
178            private static synchronized String formatTime(double time, String unit) {
179                    if (time < TEN) {
180                            return UNDER_TEN_FORMAT.format(time) + unit;
181                    }
182                    if (time < HUNDRED) {
183                            return UNDER_HUNDRED_FORMAT.format(time) + unit;
184                    }
185                    return DEFAULT_FORMAT.format(time) + unit;
186            }

188            /**
189             * Returns timestamp in human readable (yet condensed) form "yyMMdd-HHmmss.SSS".
190             *
191             * @param timestamp timestamp in millis
192             * @return timestamp as a human readable string
193             */
194            public static String presentTimestamp(long timestamp) {
195                    if (timestamp == 0) {
196                            return UNDEF_STRING;
197                    }
198                    synchronized (TIMESTAMP_FORMAT) {
199                            return TIMESTAMP_FORMAT.format(new Date(timestamp));
200                    }
201            }

233            /**
234             * Returns multi-line string containing Simon tree starting with the specified Simon.
235             * Root Simon can be used to obtain tree with all Simons. Returns {@code null} for
236             * input value of null or for NullSimon or any Simon with name equal to null (anonymous
237             * Simons) - this is also the case when the Manager is disabled and tree for its root
238             * Simon is requested.
239             *
240             * @param simon root Simon of the output tree
241             * @return string containing the tree or null if the Simon is null Simon
242             */
243            public static String simonTreeString(Simon simon) {
244                    if (simon == null || simon.getName() == null) {
245                            return null;
246                    }
247                    StringBuilder sb = new StringBuilder();
248                    printSimonTree(0, simon, sb);
249                    return sb.toString();
250            }
251    
252            private static void printSimonTree(int level, Simon simon, StringBuilder sb) {
253                    printSimon(level, simon, sb);
254                    for (Simon child : simon.getChildren()) {
255                            printSimonTree(level + 1, child, sb);
256                    }
257            }
258    
259            private static void printSimon(int level, Simon simon, StringBuilder sb) {
260                    for (int i = 0; i < level; i++) {
261                            sb.append("  ");
262                    }
263                    sb.append(localName(simon.getName()))
264                            .append('(')
265                            .append(simon.isEnabled() ? '+' : '-')
266                            .append("): ")
267                            .append(simon.toString())
268                            .append('\n');
269            }

271            /**
272             * Returns last part of Simon name - local name.
273             *
274             * @param name full Simon name
275             * @return string containing local name
276             */
277            public static String localName(String name) {
278                    int ix = name.lastIndexOf(Manager.HIERARCHY_DELIMITER);
279                    if (ix == -1) {
280                            return name;
281                    }
282                    return name.substring(ix + 1);
283            }

351            // method is extracted, so the stack trace index is always right
352            private static String generatePrivate(String suffix, boolean includeMethodName) {
353                    StackTraceElement stackElement = Thread.currentThread().getStackTrace()[CLIENT_CODE_STACK_INDEX];
354                    StringBuilder nameBuilder = new StringBuilder(stackElement.getClassName());
355                    if (includeMethodName) {
356                            nameBuilder.append('.').append(stackElement.getMethodName());
357                    }
358                    if (suffix != null) {
359                            nameBuilder.append(suffix);
360                    }
361                    return nameBuilder.toString();
362            }

374            /**
375             * Calls a block of code with stopwatch around and returns result.
376             *
377             * @param name name of the Stopwatch
378             * @param callable callable block of code
379             * @param <T> return type
380             * @return whatever block of code returns
381             * @throws Exception whatever block of code throws
382             * @since 3.0
383             */
384            public static <T> T doWithStopwatch(String name, Callable<T> callable) throws Exception {
385                    Split split = SimonManager.getStopwatch(name).start();
386                    try {
387                            return callable.call();
388                    } finally {
389                            split.stop();
390                    }
391            }
392    
393            /**
394             * Calls a block of code with stopwatch around, can not return any result or throw an exception
395             * (use {@link #doWithStopwatch(String, java.util.concurrent.Callable)} instead).
396             *
397             * @param name name of the Stopwatch
398             * @param runnable wrapped block of code
399             * @since 3.0
400             */
401            public static void doWithStopwatch(String name, Runnable runnable) {
402                    Split split = SimonManager.getStopwatch(name).start();
403                    try {
404                            runnable.run();
405                    } finally {
406                            split.stop();
407                    }
408            }

410            private static final String SHRINKED_STRING = "...";
411    
412            /**
413             * Shrinks the middle of the input string if it is too long, so it does not exceed limitTo.
414             *
415             * @since 3.2
416             */
417            public static String compact(String input, int limitTo) {
418                    if (input == null || input.length() <= limitTo) {
419                            return input;
420                    }
421    
422                    int headLength = limitTo / 2;
423                    int tailLength = limitTo - SHRINKED_STRING.length() - headLength;
424                    if (tailLength < 0) {
425                            tailLength = 1;
426                    }
427    
428                    return input.substring(0, headLength) + SHRINKED_STRING + input.substring(input.length() - tailLength);
429            }

432             * Aggregate statistics from all stopwatches in hierarchy of simons.
433             *
434             * @param simon root of the hierarchy of simons for which statistics will be aggregated
435             * @return aggregated statistics
436             * @since 3.5
437             */
438            public static StopwatchAggregate calculateStopwatchAggregate(Simon simon) {
439                    return calculateStopwatchAggregate(simon, null);
440            }

442            /**
443             * Aggregate statistics from all stopwatches in hierarchy that pass specified filter. Filter is applied
444             * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
445             * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
446             *
447             * @param simon root of the hierarchy of simons for which statistics will be aggregated
448             * @param filter filter to select subsets of simons to aggregate
449             * @return aggregates statistics
450             * @since 3.5
451             */
452            public static StopwatchAggregate calculateStopwatchAggregate(Simon simon, SimonFilter filter) {
453                    StopwatchAggregate stopwatchAggregate = new StopwatchAggregate();
454                    if (filter != null) {
455                            aggregateStopwatches(stopwatchAggregate, simon, filter);
456                    } else {
457                            aggregateStopwatches(stopwatchAggregate, simon, ACCEPT_ALL_FILTER);
458                    }
459    
460                    return stopwatchAggregate;
461            }
462    
463            private static void aggregateStopwatches(StopwatchAggregate aggregate, Simon simon, SimonFilter filter) {
464                    if (filter.accept(simon)) {
465                            if (simon instanceof Stopwatch) {
466                                    Stopwatch stopwatch = (Stopwatch) simon;
467                                    aggregate.addSample(stopwatch.sample());
468                            }
469    
470                            for (Simon child : simon.getChildren()) {
471                                    aggregateStopwatches(aggregate, child, filter);
472                            }
473                    }
474            }

487            /**
488             * Aggregate statistics from all counters in hierarchy that pass specified filter. Filter is applied
489             * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
490             * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
491             *
492             * @param simon root of the hierarchy of simons for which statistics will be aggregated
493             * @param filter filter to select subsets of simons to aggregate
494             * @return aggregates statistics
495             * @since 3.5
496             */
497            public static CounterAggregate calculateCounterAggregate(Simon simon, SimonFilter filter) {
498                    CounterAggregate aggregate = new CounterAggregate();
499                    if (filter != null) {
500                            aggregateCounters(aggregate, simon, filter);
501                    } else {
502                            aggregateCounters(aggregate, simon, ACCEPT_ALL_FILTER);
503                    }
504                    return aggregate;
505            }
506    
507            private static void aggregateCounters(CounterAggregate aggregate, Simon simon, SimonFilter filter) {
508                    if (filter.accept(simon)) {
509                            if (simon instanceof Counter) {
510                                    Counter counter = (Counter) simon;
511                                    aggregate.addSample(counter.sample());
512                            }
513    
514                            for (Simon child : simon.getChildren()) {
515                                    aggregateCounters(aggregate, child, filter);
516                            }
517                    }
518            }
