001    package org.javasimon.console;
002    
003    /**
004     * Strategies used to to display times
005     *
006     * @author gquintana
007     */
008    public enum TimeFormatType {
009            NANOSECOND(1L),
010            MICROSECOND(1000L),
011            MILLISECOND(1000000L),
012            SECOND(     1000000000L),
013            AUTO(       0L);
014    
015            private final long longFactor;
016            private final double doubleFactor;
017    
018            private TimeFormatType(long longFactor) {
019                    this.longFactor = longFactor;
020                    this.doubleFactor=(double) longFactor;
021            }
022            
023    
024            public double convert(double value) {
025                    return value / doubleFactor;
026            }
027    
028            public long convert(long value) {
029                    return value / longFactor;
030            }
031    }
