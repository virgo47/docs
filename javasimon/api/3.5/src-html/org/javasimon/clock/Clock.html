001    package org.javasimon.clock;
002    
003    /**
004     * Interface for getting current time. By default {@link Clock#SYSTEM} is used to match System timers,
005     * but different implementation can be used to provide different time for {@link org.javasimon.Manager}.
006     * This also allows easier testing without sleeping.
007     *
008     * @author <a href="mailto:ivan.mushketyk@gmail.com">Ivan Mushketyk</a>
009     * @since 3.5
010     */
011    public interface Clock {
012    
013            /**
014             * Gets current time in nanoseconds.
015             *
016             * @return current time in nanoseconds
017             */
018            long nanoTime();
019    
020            /**
021             * Gets current time in milliseconds.
022             *
023             * @return current time in milliseconds
024             */
025            long milliTime();
026    
027            /**
028             * Converts nano timer value into millis timestamp compatible with {@link #milliTime()} without actually
029             * using the timer. Method does not just divide nanos by one million, but also works with remembered
030             * values for milli- and nano-timers at one particular moment.
031             *
032             * @param nanos nano timer value
033             * @return ms timestamp
034             * @since 3.5 (before in SimonUtils since 3.3, before in SimonManager since 3.1)
035             */
036            long millisForNano(long nanos);
037    
038            /** Default implementation using {@link java.lang.System#currentTimeMillis()} and {@link java.lang.System#nanoTime()}. */
039            Clock SYSTEM = new Clock() {
040                    @Override
041                    public long nanoTime() {
042                            return System.nanoTime();
043                    }
044    
045                    @Override
046                    public long milliTime() {
047                            return System.currentTimeMillis();
048                    }
049    
050                    @Override
051                    public long millisForNano(long nanos) {
052                            return ClockUtils.millisForNano(nanos);
053                    }
054            };
055    
056            /** Clock implementation measuring CPU time. */
057            Clock CPU = new CpuClock();
058    }
