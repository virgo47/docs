001    package org.javasimon.callback.quantiles;
002    
003    import org.javasimon.Stopwatch;
004    
005    /** Enumeration of buckets types, used for configuration purposes. */
006    public enum BucketsType {
007            LINEAR() {
008                    public Buckets createBuckets(Stopwatch stopwatch, long min, long max, int bucketNb) {
009                            return new LinearBuckets(min, max, bucketNb);
010                    }
011            },
012            EXPONENTIAL() {
013                    public Buckets createBuckets(Stopwatch stopwatch, long min, long max, int bucketNb) {
014                            return new ExponentialBuckets(min, max, bucketNb);
015                    }
016            };
017    
018            /** Factory method to create {@link Buckets}. */
019            public abstract Buckets createBuckets(Stopwatch stopwatch, long min, long max, int bucketNb);
020    }
