001    package org.javasimon.jmx;
002    
003    import org.javasimon.StopwatchSample;
004    
005    import java.util.List;
006    
007    /**
008     * Interface of Simon management bean (MXBean) representing single point of access to a particular Simon
009     * {@link org.javasimon.Manager}.
010     * <p/>
011     * It provides general management tasks over all Simons for a single Manager (hierarchy of Simons)
012     * plus some useful util functions.
013     *
014     * @author Radovan Sninsky
015     * @since 2.0
016     */
017    @SuppressWarnings("UnusedDeclaration")
018    public interface SimonManagerMXBean {
019    
020            /**
021             * Enables the Simon Manager, enable monitoring application by Simons.
022             *
023             * @see org.javasimon.Manager#enable()
024             */
025            void enable();
026    
027            /**
028             * Disables the Simon Manager, disable monitoring application by Simons.
029             *
030             * @see org.javasimon.Manager#disable()
031             */
032            void disable();
033    
034            /**
035             * Returns true if the Simon Manager is enabled, if monitoring is enabled.
036             *
037             * @return true if the Simon Manager is enabled
038             * @see org.javasimon.Manager#isEnabled()
039             */
040            boolean isEnabled();
041    
042            /**
043             * Returns array containing full hierarchical names of all existing Simons.
044             *
045             * @return array of all Simon names
046             * @see org.javasimon.Manager#getSimonNames()
047             */
048            String[] getSimonNames();
049    
050            /**
051             * Returns array containing full hierarchical names of all existing Simons in natural String order.
052             *
053             * @return array of all Simon names (ordered)
054             * @see org.javasimon.Manager#getSimonNames()
055             */
056            String[] getSimonNamesOrdered();
057    
058            /**
059             * Returns type of Simon, either COUNTER, STOPWATCH or UNKNOWN.
060             *
061             * @param name name of Simon
062             * @return string COUNTER if Counter Simon, STOPWATCH if Stopwatch Simon
063             * or UNKNOWN if there is no Simon just undefined hierarchy node
064             */
065            String getType(String name);
066    
067            /**
068             * Returns array containing names and types of all existing Simons ordered naturally by name.
069             *
070             * @return array of {@link SimonInfo} objects
071             */
072            SimonInfo[] getSimonInfos();
073    
074            /**
075             * Clears the Manager (ignored if manager is disabled). All Simons are lost,
076             * but the configuration is preserved.
077             *
078             * @see org.javasimon.Manager#clear()
079             */
080            void clearManager();
081    
082            /**
083             * Enables particular Simon only.
084             *
085             * @param name name of the Simon
086             * @see org.javasimon.Simon#setState(org.javasimon.SimonState, boolean)
087             */
088            void enableSimon(String name);
089    
090            /**
091             * Disables particular Simon only.
092             *
093             * @param name name of the Simon
094             * @see org.javasimon.Simon#setState(org.javasimon.SimonState, boolean)
095             */
096            void disableSimon(String name);
097    
098            /**
099             * Lets the Simon to inherit its enable/disable state from its parent.
100             *
101             * @param name name of the Simon
102             */
103            void inheritState(String name);
104    
105            /**
106             * Retrieves sample data object for a particular Counter.
107             *
108             * @param name name of the Simon
109             * @return sample object or null if Simon with entered name doesn't exist
110             * @see org.javasimon.CounterSample
111             */
112            CounterSample getCounterSample(String name);
113    
114            /**
115             * Retrieves sample data object for a particular Stopwatch.
116             *
117             * @param name name of the Simon
118             * @return sample object or null if Simon with entered name doesn't exist
119             * @see org.javasimon.StopwatchSample
120             */
121            StopwatchSample getStopwatchSample(String name);
122    
123            /**
124             * Retrieves sample data object for a particular Counter and resets it.
125             *
126             * @param name name of the Simon
127             * @return sample object or null if Simon with entered name doesn't exist
128             * @see org.javasimon.CounterSample
129             * @deprecated use {@link org.javasimon.jmx.SimonManagerMXBean#getIncrementCounterSample(String, String)}
130             */
131            @Deprecated
132            CounterSample getCounterSampleAndReset(String name);
133    
134            /**
135             * Samples increment in Counter values since the previous call of this method with the
136             * same key. When the method is called the first time for the key, current values
137             * are returned (same like from {@link #getCounterSample(String)} }. Any subsequent calls with the key
138             * provide increments.
139             * <p/>
140             * Clients should use a unique key (GUID, host name, etc.), to avoid interference
141             * with other clients.
142             *
143             * @param name name of the stopwatch
144             * @param key name of an incremental sample
145             * @return sample collected for a specified key if it was created before
146             */
147            CounterSample getIncrementCounterSample(String name, String key);
148    
149            /**
150             * Retrieves sample data object for a particular Stopwatch and resets it.
151             *
152             * @param name name of the Simon
153             * @return sample object or null if Simon with entered name doesn't exist
154             * @see org.javasimon.StopwatchSample
155             * @deprecated use {@link #getIncrementStopwatchSample(String, String)}  instead.
156             */
157            @Deprecated
158            StopwatchSample getStopwatchSampleAndReset(String name);
159    
160    
161            /**
162             * Samples increment in Stopwatch values since the previous call of this method with the
163             * same key. When the method is called the first time for the key, current values
164             * are returned (same like from {@link #getCounterSample(String)} }. Any subsequent calls with the key
165             * provide increments.
166             * <p/>
167             * Clients should use a unique key (GUID, host name, etc.), to avoid interference
168             * with other clients.
169             *
170             * @param name name of the stopwatch
171             * @param key name of an incremental sample
172             * @return sample collected for a specified key if it was created before
173             */
174            org.javasimon.jmx.StopwatchSample getIncrementStopwatchSample(String name, String key);
175    
176            /**
177             * Resets the particular Simon by its name.
178             *
179             * @param name name of the Simon
180             * @deprecated will be removed in 4.0
181             */
182            @Deprecated
183            void reset(String name);
184    
185            /**
186             * Prints multi-line string containing Simon tree starting with the specified Simon to standard output.
187             *
188             * @see org.javasimon.utils.SimonUtils#simonTreeString(org.javasimon.Simon)
189             */
190            void printSimonTree();
191    
192            /**
193             * Sample all Counters
194             *
195             * @return One Sample for each Counter
196             */
197            List<CounterSample> getCounterSamples();
198    
199            /**
200             * Increment sample all Counters.
201             *
202             * @param key name of an incremental sample
203             * @return one Sample for each Counter
204             * @see #getIncrementCounterSample(String, String)
205             */
206            List<CounterSample> getIncrementCounterSamples(String key);
207    
208            /**
209             * Sample all Counters whose name matches given pattern
210             *
211             * @param namePattern Name pattern, null means all Counters
212             * @return One Sample for each Counter
213             */
214            List<CounterSample> getCounterSamples(String namePattern);
215    
216            /**
217             * Incremental sample for all Counters whose name matches given pattern.     *
218             *
219             * @param namePattern name pattern ({@link org.javasimon.SimonPattern}), {@code null} means all Counters
220             * @param key name of an incremental sample
221             * @return one Sample for each Counter whose name matches given pattern
222             * @see #getIncrementCounterSample(String, String)
223             */
224            List<CounterSample> getIncrementCounterSamples(String namePattern, String key);
225    
226            /**
227             * Sample all Stopwatches whose name matches given pattern.
228             *
229             * @param namePattern name pattern ({@link org.javasimon.SimonPattern}), {@code null} means all Stopwatches
230             * @return one Sample for each Stopwatch
231             */
232            List<org.javasimon.jmx.StopwatchSample> getStopwatchSamples(String namePattern);
233    
234            /**
235             * Increment sample all Stopwatches whose name matches given pattern.
236             *
237             * @param namePattern name pattern ({@link org.javasimon.SimonPattern}), {@code null} means all Stopwatches
238             * @param key name of an incremental sample
239             * @return one Sample for each Stopwatch whose name matches given pattern
240             * @see #getIncrementStopwatchSample(String, String)
241             */
242            List<org.javasimon.jmx.StopwatchSample> getIncrementStopwatchSamples(String namePattern, String key);
243    
244            /**
245             * Sample all Stopwatches.
246             *
247             * @return one Sample for each Stopwatch
248             */
249            List<org.javasimon.jmx.StopwatchSample> getStopwatchSamples();
250    
251            /**
252             * Increment sample all Stopwatches.
253             *
254             * @param key name of an incremental sample
255             * @return one Sample for each Stopwatch
256             * @see #getIncrementStopwatchSample(String, String)
257             */
258            List<org.javasimon.jmx.StopwatchSample> getIncrementStopwatchSamples(String key);
259    }
