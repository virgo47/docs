001    package org.javasimon.jmx;
002    
003    import org.javasimon.Sample;
004    
005    import java.util.List;
006    
007    /**
008     * Interface with common methods for JMX beans for a single Simon that corresponds
009     * to AbstractSimon in the core package.
010     *
011     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
012     */
013    public interface SimonSuperMXBean {
014    
015            /**
016             * Returns Simon name. While Simon names can be {@code null} for anonymous
017             * Simons, such Simons are never registered hence this method never returns
018             * {@code null}.
019             *
020             * @return name of the Simon
021             */
022            String getName();
023    
024            /**
025             * Returns name of the parent Simon.
026             *
027             * @return name of the parent Simon
028             */
029            String getParentName();
030    
031            /**
032             * Returns list of children names.
033             *
034             * @return list of children names
035             */
036            List<String> getChildrenNames();
037    
038            /**
039             * Returns state of the Simon that can be ENABLED, DISABLED or INHERITED. State is returned
040             * as a String that matches values of the {@link org.javasimon.SimonState} enumeration.
041             *
042             * @return state of the Simon as a String
043             */
044            String getState();
045    
046            /**
047             * Sets the state of the Simon. It must be specified whether to propagate the change
048             * and overrule states of all sub-simons which effectively sets the same state to the whole
049             * subtree. If subtree is not overruled, some Simons (with their subtrees) may not be affected
050             * by this change.
051             *
052             * @param state a new state as a valid String for {@link org.javasimon.SimonState#valueOf(String)}
053             * @param overrule specifies whether this change is forced to the whole subtree.
054             */
055            void setState(String state, boolean overrule);
056    
057            /**
058             * Returns true, if the Simon is enabled or if the enabled state is inherited.
059             *
060             * @return true, if the Simon is effectively enabled
061             */
062            boolean isEnabled();
063    
064            /**
065             * Resets the Simon, its usages and stat processor - concrete values depend
066             * on the type and the implementation.
067             *
068             * @deprecated will be removed in 4.0
069             */
070            @Deprecated
071            void reset();
072    
073            /**
074             * Returns note for the Simon. Note enables Simon with an additional information in human
075             * readable form.
076             *
077             * @return note for the Simon.
078             */
079            String getNote();
080    
081            /**
082             * Sets note for the Simon. Note enables Simon with an additional information in human
083             * readable form.
084             *
085             * @param note note for the Simon.
086             */
087            void setNote(String note);
088    
089            /**
090             * Returns ms timestamp of the first usage of this Simon. First and last usage
091             * are updated when monitor performs the measuring (start/stop/count/etc). They
092             * are not updated when values are obtained from the monitor.
093             *
094             * @return ms timestamp of the first usage
095             */
096            long getFirstUsage();
097    
098            /**
099             * Returns nicely formatted timestamp of the first usage of this Simon.
100             *
101             * @return formatted date and time of the first usage
102             */
103            String getFirstUsageAsString();
104    
105            /**
106             * Returns ms timestamp of the last usage of this Simon. First and last usage
107             * are updated when monitor performs the measuring (start/stop/count/etc). They
108             * are not updated when values are obtained from the monitor.
109             *
110             * @return ms timestamp of the last usage
111             */
112            long getLastUsage();
113    
114            /**
115             * Returns nicely formatted timestamp of the last usage of this Simon.
116             *
117             * @return formatted date and time of the last usage
118             */
119            String getLastUsageAsString();
120    
121            /**
122             * Samples Simon values and returns them in a Java Bean derived from Sample interface.
123             *
124             * @return sample containing all Simon values
125             */
126            Sample sample();
127    
128            /**
129             * Samples Simon values and returns them in a Java Bean derived from Sample interface
130             * and resets the Simon. Operation is synchronized to assure atomicity.
131             *
132             * @return sample containing all Simon values
133             * @deprecated use {@link #sampleIncrement(String)}
134             */
135            @Deprecated
136            Sample sampleAndReset();
137    
138            /**
139             * Samples increment in Simon values since the previous call of this method with the
140             * same key. When the method is called the first time for the key, current values
141             * are returned (same like from {@link #sample()}. Any subsequent calls with the key
142             * provide increments.
143             * <p/>
144             * Clients should use a unique key (GUID, host name, etc.), to avoid interference
145             * with other clients.
146             *
147             * @param key key used to access incremental sample
148             * @return {@link org.javasimon.Sample} with value increments
149             */
150            Sample sampleIncrement(String key);
151    
152            /**
153             * Returns Simon type used as a property in the {@link javax.management.ObjectName}.
154             *
155             * @return Simon type
156             */
157            String getType();
158    
159            /**
160             * Stops incremental sampling for the key, removing any internal information for the key.
161             * Next call to {@link #sampleIncrement(String)} for the key will be considered the first
162             * call for the key.
163             *
164             * @param key key used to access incremental sample
165             * @return true if incremental information for the key existed, false otherwise
166             */
167            boolean stopIncrementalSampling(String key);
168    }
