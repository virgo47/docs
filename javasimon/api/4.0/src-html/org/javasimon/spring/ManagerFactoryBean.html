001package org.javasimon.spring;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.javasimon.Manager;
007import org.javasimon.SimonManager;
008import org.javasimon.SwitchingManager;
009import org.javasimon.callback.Callback;
010
011import org.springframework.beans.factory.FactoryBean;
012
013/**
014 * Factory bean which produces Simon manager.
015 * Spring configuration:
016 * <pre>{@code
017 * <bean id="jmxServer" class="org.springframework.jmx.support.MBeanServerFactoryBean">
018 *     <property name="locateExistingServerIfPossible" value="true"/>
019 * </bean>
020 * <bean id="simonManager" class="org.javasimon.spring.ManagerFactoryBean">
021 *     <property name="callbacks">
022 *         <list>
023 *             <bean class="org.javasimon.jmx.JmxRegisterCallback">
024 *                 <constructor-arg ref="jmxServer"/>
025 *                 <constructor-arg value="org.javasimon"/>
026 *             </bean>
027 *             <bean class="org.javasimon.utils.LoggingCallback"/>
028 *         </list>
029 *     </property>
030 * </bean>
031 * <bean id="simonInterceptor" class="org.javasimon.spring.MonitoringInterceptor">
032 *     <property name="manager" ref="simonManager"/>
033 * </bean>}
034 * </pre>
035 *
036 * @author gquintana
037 */
038public class ManagerFactoryBean implements FactoryBean<Manager> {
039        /**
040         * Flag indicating whether Simon manager singleton should be used.
041         * Defaults to true.
042         */
043        private boolean singleton = true;
044        /**
045         * List of callbacks to add to the manager
046         */
047        private List<Callback> callbacks = new ArrayList<>();
048        /**
049         * Flag indicated whether Simon manager should enabled or not.
050         * Defaults to true.
051         */
052        private boolean enabled = true;
053
054        /**
055         * Get simon manager instance.
056         * If singleton is enabled SimonManager.manager() is invoked else new
057         * Manager is created.
058         * Then callbacks are appended to this manager
059         *
060         * @return Simon manager
061         */
062        public Manager getObject() throws Exception {
063                Manager manager;
064                if (singleton) {
065                        manager = SimonManager.manager();
066                } else {
067                        manager = new SwitchingManager();
068                }
069                registerCallbacks(manager);
070                configureEnabled(manager);
071                return manager;
072        }
073
074        /**
075         * @return Always Manager class
076         */
077        public Class<?> getObjectType() {
078                return Manager.class;
079        }
080
081        public boolean isSingleton() {
082                return singleton;
083        }
084
085        public void setSingleton(boolean singleton) {
086                this.singleton = singleton;
087        }
088
089        /**
090         * Register callbacks in given manager
091         *
092         * @param manager Manager
093         */
094        private void registerCallbacks(Manager manager) {
095                if (callbacks != null) {
096                        for (Callback callback : callbacks) {
097                                manager.callback().addCallback(callback);
098                        }
099                }
100        }
101
102        public List<Callback> getCallbacks() {
103                return callbacks;
104        }
105
106        public void setCallbacks(List<Callback> callbacks) {
107                this.callbacks = callbacks;
108        }
109
110        /**
111         * When needed toggle the enabled flag of given Simon manager
112         *
113         * @param manager Simon manager
114         */
115        private void configureEnabled(Manager manager) {
116                if (enabled != manager.isEnabled()) {
117                        if (enabled) {
118                                manager.enable();
119                        } else {
120                                manager.disable();
121                        }
122                }
123        }
124
125        public boolean isEnabled() {
126                return enabled;
127        }
128
129        public void setEnabled(boolean enabled) {
130                this.enabled = enabled;
131        }
132}
