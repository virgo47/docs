020/**
021 * SimonUtils provides static utility methods.
022 * <p/>
023 * <h3>Human readable outputs</h3>
024 * Both {@link org.javasimon.Stopwatch} and {@link org.javasimon.Counter} provide human readable
025 * {@code toString} outputs. All nanosecond values are converted into few valid digits with
026 * proper unit (ns, us, ms, s) - this is done via method {@link #presentNanoTime(long)}.
027 * Max/min counter values are checked for undefined state (max/min long value is converted
028 * to string "undef") - via method {@link #presentMinMaxCount(long)}.
029 * <p/>
030 * <h3>Aggregation utilities</h3>
031 * It is possible to sum up (aggregate) values for a subtree for a particular Simon type using
032 * {@link #calculateCounterAggregate(org.javasimon.Simon)} or {@link #calculateStopwatchAggregate(org.javasimon.Simon)}.
033 * Methods come also in versions allowing to filter by {@link org.javasimon.SimonFilter}.
034 * <p/>
035 * <h3>Simon tree operations</h3>
036 * For various debug purposes there is a method that creates string displaying the whole Simon sub-tree.
037 * Here is example code that initializes two random Simons and prints the whole Simon hierarchy
038 * (note that the method can be used to obtain any sub-tree of the hierarchy):
039 * <pre>
040 * Split split = SimonManager.getStopwatch("com.my.other.stopwatch").start();
041 * SimonManager.getCounter("com.my.counter").setState(SimonState.DISABLED, false);
042 * split.stop();
043 * System.out.println(SimonUtils.simonTreeString(SimonManager.getRootSimon()));</pre>
044 * And the output is:
045 * <pre>
046 * (+): Unknown Simon:  [ ENABLED]
047 *   com(+): Unknown Simon:  [com INHERIT]
048 *     my(+): Unknown Simon:  [com.my INHERIT]
049 *       other(+): Unknown Simon:  [com.my.other INHERIT]
050 *         stopwatch(+): Simon Stopwatch: total 24.2 ms, counter 1, max 24.2 ms, min 24.2 ms, mean 24.2 ms [com.my.other.stopwatch INHERIT]
051 *       counter(-): Simon Counter: counter=0, max=undef, min=undef [com.my.counter DISABLED]</pre>
052 * Notice +/- signs in parenthesis that displays effective Simon state (enabled/disabled), further
053 * details are printed via each Simon's {@code toString} method.
054 *
055 * <h3>Other utilities</h3>
056 * It is possible to obtain "local name" of the Simon (behind the last dot) via {@link #localName(String)}
057 * or check if the name is valid Simon name via {@link #checkName(String)}.
058 *
059 * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
060 * @author Radovan Sninsky
061 * @since 1.0
062 */

103        static {
104                // Finds out the index of "this code" in the returned stack trace - funny but it differs in JDK 1.5 and 1.6
105                int i = 1;
106                for (StackTraceElement ste : Thread.currentThread().getStackTrace()) {
107                        i++;
108                        if (ste.getClassName().equals(SimonUtils.class.getName())) {
109                                break;
110                        }
111                }
112                CLIENT_CODE_STACK_INDEX = i;
113        }

126        /**
127         * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
128         * except for seconds that are the biggest unit used. In case of undefined value ({@link Long#MAX_VALUE} or
129         * {@link Long#MIN_VALUE}) string {@link #UNDEF_STRING} is returned, so it can be used for min/max values as well.
130         *
131         * @param nanos time in nanoseconds
132         * @return human readable time string
133         */
134        public static String presentNanoTime(long nanos) {
135                if (nanos == Long.MAX_VALUE || nanos == Long.MIN_VALUE) {
136                        return UNDEF_STRING;
137                }
138                return presentNanoTimePrivate((double) nanos);
139        }
140
141        /**
142         * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
143         * except for seconds that are the biggest unit used.
144         *
145         * @param nanos time in nanoseconds
146         * @return human readable time string
147         */
148        public static String presentNanoTime(double nanos) {
149                if (nanos == Double.MAX_VALUE) {
150                        return UNDEF_STRING;
151                }
152                return presentNanoTimePrivate(nanos);
153        }
154
155        private static String presentNanoTimePrivate(double time) {
156                if (Math.abs(time) < 1d) {
157                        return "0";
158                }
159
160                if (time < UNIT_PREFIX_FACTOR) {
161                        return ((long) time) + " ns";
162                }
163
164                time /= UNIT_PREFIX_FACTOR;
165                if (time < UNIT_PREFIX_FACTOR) {
166                        return formatTime(time, " us");
167                }
168
169                time /= UNIT_PREFIX_FACTOR;
170                if (time < UNIT_PREFIX_FACTOR) {
171                        return formatTime(time, " ms");
172                }
173
174                time /= UNIT_PREFIX_FACTOR;
175                return formatTime(time, " s");
176        }
177
178        private static synchronized String formatTime(double time, String unit) {
179                if (time < TEN) {
180                        return UNDER_TEN_FORMAT.format(time) + unit;
181                }
182                if (time < HUNDRED) {
183                        return UNDER_HUNDRED_FORMAT.format(time) + unit;
184                }
185                return DEFAULT_FORMAT.format(time) + unit;
186        }

188        /**
189         * Returns timestamp in human readable (yet condensed) form "yyMMdd-HHmmss.SSS".
190         *
191         * @param timestamp timestamp in millis
192         * @return timestamp as a human readable string
193         */
194        public static String presentTimestamp(long timestamp) {
195                if (timestamp == 0) {
196                        return UNDEF_STRING;
197                }
198                synchronized (TIMESTAMP_FORMAT) {
199                        return TIMESTAMP_FORMAT.format(new Date(timestamp));
200                }
201        }

203        /**
204         * Returns min/max counter values in human readable form - if the value is max or min long value
205         * it is considered unused and string "undef" is returned.
206         *
207         * @param minmax counter extreme value
208         * @return counter value or "undef" if counter contains {@code Long.MIN_VALUE} or {@code Long.MAX_VALUE}

210        public static String presentMinMaxCount(long minmax) {
211                if (minmax == Long.MAX_VALUE || minmax == Long.MIN_VALUE) {
212                        return UNDEF_STRING;

217        /**
218         * Returns multi-line string containing Simon tree starting with the specified Simon.
219         * Root Simon can be used to obtain tree with all Simons. Returns {@code null} for
220         * input value of null or for NullSimon or any Simon with name equal to null (anonymous
221         * Simons) - this is also the case when the Manager is disabled and tree for its root
222         * Simon is requested.
223         *
224         * @param simon root Simon of the output tree
225         * @return string containing the tree or null if the Simon is null Simon
226         */
227        public static String simonTreeString(Simon simon) {
228                if (simon == null || simon.getName() == null) {
229                        return null;
230                }
231                StringBuilder sb = new StringBuilder();
232                printSimonTree(0, simon, sb);
233                return sb.toString();
234        }
235
236        private static void printSimonTree(int level, Simon simon, StringBuilder sb) {
237                printSimon(level, simon, sb);
238                for (Simon child : simon.getChildren()) {
239                        printSimonTree(level + 1, child, sb);
240                }
241        }
242
243        private static void printSimon(int level, Simon simon, StringBuilder sb) {
244                for (int i = 0; i < level; i++) {
245                        sb.append("  ");
246                }
247                sb.append(localName(simon.getName()))
248                        .append('(')
249                        .append(simon.isEnabled() ? '+' : '-')
250                        .append("): ")
251                        .append(simon.toString())
252                        .append('\n');
253        }

255        /**
256         * Returns last part of Simon name - local name.
257         *
258         * @param name full Simon name
259         * @return string containing local name
260         */
261        public static String localName(String name) {
262                int ix = name.lastIndexOf(Manager.HIERARCHY_DELIMITER);
263                if (ix == -1) {
264                        return name;
265                }
266                return name.substring(ix + 1);
267        }

318        // method is extracted, so the stack trace index is always right
319        private static String generatePrivate(String suffix, boolean includeMethodName) {
320                StackTraceElement stackElement = Thread.currentThread().getStackTrace()[CLIENT_CODE_STACK_INDEX];
321                StringBuilder nameBuilder = new StringBuilder(stackElement.getClassName());
322                if (includeMethodName) {
323                        nameBuilder.append('.').append(stackElement.getMethodName());
324                }
325                if (suffix != null) {
326                        nameBuilder.append(suffix);
327                }
328                return nameBuilder.toString();
329        }

342         * Calls a block of code with stopwatch around and returns result.
343         *
344         * @param name name of the Stopwatch
345         * @param callable callable block of code

352                try (Split split = SimonManager.getStopwatch(name).start()) {

354                }

371        private static final String SHRINKED_STRING = "...";
372
373        /**
374         * Shrinks the middle of the input string if it is too long, so it does not exceed limitTo.
375         *
376         * @since 3.2
377         */
378        public static String compact(String input, int limitTo) {
379                if (input == null || input.length() <= limitTo) {
380                        return input;
381                }
382
383                int headLength = limitTo / 2;
384                int tailLength = limitTo - SHRINKED_STRING.length() - headLength;
385                if (tailLength < 0) {
386                        tailLength = 1;
387                }
388
389                return input.substring(0, headLength) + SHRINKED_STRING + input.substring(input.length() - tailLength);
390        }

395         * @param simon root of the hierarchy of simons for which statistics will be aggregated
396         * @return aggregated statistics
397         * @since 3.5
398         */
399        public static StopwatchAggregate calculateStopwatchAggregate(Simon simon) {
400                return calculateStopwatchAggregate(simon, null);
401        }

403        /**
404         * Aggregate statistics from all stopwatches in hierarchy that pass specified filter. Filter is applied
405         * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
406         * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
407         *
408         * @param simon root of the hierarchy of simons for which statistics will be aggregated
409         * @param filter filter to select subsets of simons to aggregate
410         * @return aggregates statistics
411         * @since 3.5
412         */
413        public static StopwatchAggregate calculateStopwatchAggregate(Simon simon, SimonFilter filter) {
414                StopwatchAggregate stopwatchAggregate = new StopwatchAggregate();
415                if (filter != null) {
416                        aggregateStopwatches(stopwatchAggregate, simon, filter);
417                } else {
418                        aggregateStopwatches(stopwatchAggregate, simon, ACCEPT_ALL_FILTER);
419                }
420
421                return stopwatchAggregate;
422        }
423
424        private static void aggregateStopwatches(StopwatchAggregate aggregate, Simon simon, SimonFilter filter) {
425                if (filter.accept(simon)) {
426                        if (simon instanceof Stopwatch) {
427                                Stopwatch stopwatch = (Stopwatch) simon;
428                                aggregate.addSample(stopwatch.sample());
429                        }
430
431                        for (Simon child : simon.getChildren()) {
432                                aggregateStopwatches(aggregate, child, filter);
433                        }
434                }
435        }

448        /**
449         * Aggregate statistics from all counters in hierarchy that pass specified filter. Filter is applied
450         * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
451         * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
452         *
453         * @param simon root of the hierarchy of simons for which statistics will be aggregated
454         * @param filter filter to select subsets of simons to aggregate
455         * @return aggregates statistics
456         * @since 3.5
457         */
458        public static CounterAggregate calculateCounterAggregate(Simon simon, SimonFilter filter) {
459                CounterAggregate aggregate = new CounterAggregate();
460                if (filter != null) {
461                        aggregateCounters(aggregate, simon, filter);
462                } else {
463                        aggregateCounters(aggregate, simon, ACCEPT_ALL_FILTER);
464                }
465                return aggregate;
466        }
467
468        private static void aggregateCounters(CounterAggregate aggregate, Simon simon, SimonFilter filter) {
469                if (filter.accept(simon)) {
470                        if (simon instanceof Counter) {
471                                Counter counter = (Counter) simon;
472                                aggregate.addSample(counter.sample());
473                        }
474
475                        for (Simon child : simon.getChildren()) {
476                                aggregateCounters(aggregate, child, filter);
477                        }
478                }
479        }
