003/**
004 * Interface indicating that an object is wrapping another object.
005 *
006 * @param <T> Wrapped type
007 * @author gerald
008 */
009public interface Delegating<T> {
010        /**
011         * Get wrapped object.
012         *
013         * @return Wrapped object
014         */
015        T getDelegate();
016}
