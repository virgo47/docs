001package org.javasimon.console;
002
003import java.io.IOException;
004import javax.servlet.ServletConfig;
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009
010import org.javasimon.Manager;
011import org.javasimon.utils.SimonUtils;
012
013/**
014 * Front controller servlet of Simon Web console.
015 *
016 * @author gquintana
017 */
018@SuppressWarnings("UnusedDeclaration")
019public class SimonConsoleServlet extends HttpServlet {
020        /**
021         * Serial version UID since class is Serializable.
022         */
023        public static final long serialVersionUID = 1L;
024
025        /**
026         * URL Prefix init parameter name.
027         */
028        public static final String URL_PREFIX_INIT_PARAMETER = "url-prefix";
029        /**
030         * Plugin classes init parameter name.
031         */
032        public static final String PLUGIN_CLASSES_INIT_PARAMETER = "plugin-classes";
033        private SimonConsoleRequestProcessor requestProcessor;
034
035        @Override
036        public void init(ServletConfig config) throws ServletException {
037                super.init(config);
038
039                pickUpSharedManagerIfExists(config);
040
041                String urlPrefix = config.getInitParameter(URL_PREFIX_INIT_PARAMETER);
042                initRequestProcessor(urlPrefix);
043
044                String pluginClasses = config.getInitParameter(PLUGIN_CLASSES_INIT_PARAMETER);
045                if (pluginClasses != null) {
046                        requestProcessor.getPluginManager().addPlugins(pluginClasses);
047                }
048        }
049
050        public void initRequestProcessor(String urlPrefix) {
051                if (urlPrefix == null) {
052                        urlPrefix = "";
053                } else {
054                        urlPrefix = urlPrefix.trim();
055                }
056                requestProcessor = new SimonConsoleRequestProcessor(urlPrefix);
057                requestProcessor.initActionBindings();
058        }
059
060        public SimonConsoleRequestProcessor getRequestProcessor() {
061                return requestProcessor;
062        }
063
064        private void pickUpSharedManagerIfExists(ServletConfig config) {
065                Object managerObject = config.getServletContext().getAttribute(SimonUtils.MANAGER_SERVLET_CTX_ATTRIBUTE);
066                if (managerObject != null && managerObject instanceof Manager) {
067                        requestProcessor.setManager((Manager) managerObject);
068                }
069        }
070
071        /**
072         * Handles the HTTP
073         * <code>GET</code> method.
074         *
075         * @param request servlet request
076         * @param response servlet response
077         * @throws ServletException if a servlet-specific error occurs
078         * @throws IOException if an I/O error occurs
079         */
080        @Override
081        protected void doGet(HttpServletRequest request, HttpServletResponse response)
082                throws ServletException, IOException
083        {
084                requestProcessor.processRequest(request, response);
085        }
086
087        /**
088         * Handles the HTTP
089         * <code>POST</code> method.
090         *
091         * @param request servlet request
092         * @param response servlet response
093         * @throws ServletException if a servlet-specific error occurs
094         * @throws IOException if an I/O error occurs
095         */
096        @Override
097        protected void doPost(HttpServletRequest request, HttpServletResponse response)
098                throws ServletException, IOException
099        {
100                requestProcessor.processRequest(request, response);
101        }
102
103        /**
104         * Returns a short description of the servlet.
105         *
106         * @return a String containing servlet description
107         */
108        @Override
109        public String getServletInfo() {
110                return "Short description";
111        }
112}
