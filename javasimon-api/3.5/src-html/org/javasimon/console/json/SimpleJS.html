001    package org.javasimon.console.json;
002    
003    import java.io.IOException;
004    import java.io.Writer;
005    
006    import org.javasimon.console.text.Stringifier;
007    
008    /**
009     * JavaScript simple type.
010     *
011     * @author gquintana
012     */
013    public class SimpleJS<T> extends AnyJS {
014    
015            /** Simple value. */
016            protected final T value;
017            /** Value formatter. */
018            protected final Stringifier<T> valueStringifier;
019    
020            // TODO how hidden is this?
021            /** Hidden constructor use factory methods instead. */
022            public SimpleJS(T value, Stringifier<T> valueStringifier) {
023                    this.value = value;
024                    this.valueStringifier = valueStringifier;
025            }
026    
027            public T getValue() {
028                    return value;
029            }
030    
031            public String getFormattedValue() {
032                    return valueStringifier.toString(value);
033            }
034    
035            @Override
036            public void write(Writer writer) throws IOException {
037                    writer.write(getFormattedValue());
038            }
039    }
