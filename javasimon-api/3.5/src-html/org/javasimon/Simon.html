001    package org.javasimon;
002    
003    import java.util.List;
004    
005    /**
006     * Simon interface contains common functions related to Simon management - enable/disable and hierarchy.
007     * It does not contain any real action method - these are in specific interfaces that describes
008     * purpose of the particular type of monitor.
009     *
010     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
011     * @see Manager
012     * @see Counter for Simon counting some events
013     * @see Stopwatch for Simon measuring time spans
014     */
015    public interface Simon extends HasAttributes {
016    
017            /**
018             * Returns Simon name. Simon name is always fully qualified
019             * and determines also position of the Simon in the monitor hierarchy.
020             * Simon name can be {@code null} for anonymous Simons.
021             *
022             * @return name of the Simon
023             */
024            String getName();
025    
026            /**
027             * Returns parent Simon.
028             *
029             * @return parent Simon
030             */
031            Simon getParent();
032    
033            /**
034             * Returns list of children - direct sub-simons.
035             *
036             * @return list of children
037             */
038            List<Simon> getChildren();
039    
040            /**
041             * Returns Simon's {@link Manager}.
042             *
043             * @return Simon's {@link Manager}
044             * @since 3.5
045             */
046            Manager getManager();
047    
048            /**
049             * Returns state of the Simon that can be enabled, disabled or inherited.
050             *
051             * @return state of the Simon
052             */
053            SimonState getState();
054    
055            /**
056             * Sets the state of the Simon. It must be specified whether to propagate the change
057             * and overrule states of all sub-simons which effectively sets the same state to the whole
058             * subtree. If subtree is not overruled, some Simons (with their subtrees) may not be affected
059             * by this change.
060             *
061             * @param state a new state.
062             * @param overrule specifies whether this change is forced to the whole subtree.
063             * @throws IllegalArgumentException if {@code state} is {@code null}.
064             */
065            void setState(SimonState state, boolean overrule);
066    
067            /**
068             * Returns true, if the Simon is enabled or if the enabled state is inherited.
069             *
070             * @return true, if the Simon is effectively enabled
071             */
072            boolean isEnabled();
073    
074            /**
075             * Resets the Simon values related to the measuring, timestamps and so on - usage timestamps, state,
076             * attributes are not affected. Timestamp of the last reset can be obtained by the method {@link #getLastReset()}.
077             * Reset is performed even for disabled Simons.
078             *
079             * @deprecated will be removed in 4.0 - use {@link #sampleIncrement(Object)} for similar purposes
080             */
081            @Deprecated
082            void reset();
083    
084            /**
085             * Returns ms timestamp of the last recent usage of the {@link #reset()} method on the Simon.
086             * Returns 0 if {@code reset} was not called yet. This timestamp is useful for rate measuring
087             * when reset is called on a regular basis - likely via {@link #sampleAndReset()}. While
088             * client code could store the timestamp too it is not necessary with this method.
089             *
090             * @return ms timestamp of the last reset or 0 if reset was not called yet
091             * @deprecated will be removed in 4.0
092             */
093            @Deprecated
094            long getLastReset();
095    
096            /**
097             * Returns note for the Simon. Note enables Simon with an additional information in human
098             * readable form.
099             *
100             * @return note for the Simon.
101             */
102            String getNote();
103    
104            /**
105             * Sets note for the Simon. Note enables Simon with an additional information in human
106             * readable form.
107             *
108             * @param note note for the Simon.
109             */
110            void setNote(String note);
111    
112            /**
113             * Returns ms timestamp of the first usage of this Simon. First and last usage
114             * are updated when monitor performs the measuring (start/stop/count/etc). They
115             * are not updated when values are obtained from the monitor.
116             *
117             * @return ms timestamp of the first usage
118             */
119            long getFirstUsage();
120    
121            /**
122             * Returns ms timestamp of the last usage of this Simon. First and last usage
123             * are updated when monitor performs the measuring (start/stop/count/etc). They
124             * are not updated when values are obtained from the monitor.
125             *
126             * @return ms timestamp of the last usage
127             */
128            long getLastUsage();
129    
130            /**
131             * Samples Simon values and returns them in a Java Bean derived from Sample interface.
132             *
133             * @return sample containing all Simon values
134             */
135            Sample sample();
136    
137            /**
138             * Samples Simon values and returns them in a Java Bean derived from Sample interface
139             * and resets the Simon. Operation is synchronized to assure atomicity.
140             *
141             * @return sample containing all Simon values
142             * @deprecated will be removed in 4.0 - use {@link #sampleIncrement(Object)} instead
143             */
144            @Deprecated
145            Sample sampleAndReset();
146    
147            /**
148             * Samples increment in Simon values since the previous call of this method with the
149             * same key. When the method is called the first time for the key, current values
150             * are returned (same like from {@link #sample()}. Any subsequent calls with the key
151             * provide increments.
152             * <p/>
153             * Clients can use any sampling key (any Object) which enables safe access to their own increments.
154             * Using String does not guarantee this as any client can potentially guess the key. This
155             * may or may not be an issue.
156             *
157             * @param key sampling key used to access incremental sample
158             * @return {@link org.javasimon.Sample} with value increments
159             * @since 3.5
160             */
161            Sample sampleIncrement(Object key);
162    
163            /**
164             * Stops incremental sampling for the key, removing any internal information for the key.
165             * Next call to {@link #sampleIncrement(Object)} for the key will be considered the first
166             * call for the key.
167             *
168             * @param key sampling key used to access incremental sample
169             * @return true if incremental information for the key existed, false otherwise
170             * @since 3.5
171             */
172            boolean stopIncrementalSampling(Object key);
173    }
