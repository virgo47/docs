001    package org.javasimon.jmx;
002    
003    import org.javasimon.utils.SimonUtils;
004    
005    import java.beans.ConstructorProperties;
006    import java.util.Date;
007    
008    /**
009     * Value object for retrieving data from Counter Simon. Basically, it's
010     * {@link org.javasimon.CounterSample} with added JMX capabilities to be return as object via
011     * MXBean method.
012     * <p/>
013     * Example:
014     * <pre>
015     * SimonManagerMXBean simon = JMX.newMXBeanProxy(..., new ObjectName("domain:type=Simon"), SimonManagerMXBean.class);
016     * CounterSample = simon.getCounterSample("simon.counter");
017     * </pre>
018     *
019     * @author Radovan Sninsky
020     * @since 2.0
021     */
022    public final class CounterSample extends org.javasimon.CounterSample {
023    
024            /**
025             * JMX constructor. Constructor used by JMX client code to initialize all properties of object
026             * from composite data object.
027             *
028             * @param name Simon's name
029             * @param note note (provided optionally)
030             * @param firstUsage first usage ms timestamp
031             * @param lastUsage last usage ms timestamp
032             * @param lastReset last reset ms timestamp
033             * @param counter actual counter value
034             * @param min minimal counter value
035             * @param max maximal counter value
036             * @param minTimestamp time when counter reached minimal value
037             * @param maxTimestamp time when counter reached maximal value
038             * @param incSum sum of all increments
039             * @param decSum sum of all decrements
040             */
041            @ConstructorProperties({"name", "note", "firstUsage", "lastUsage", "lastReset", "counter", "min",
042                    "max", "minTimestamp", "maxTimestamp", "incrementSum", "decrementSum"})
043            public CounterSample(String name, String note, long firstUsage, long lastUsage, long lastReset, long counter,
044                                 long min, long max, long minTimestamp, long maxTimestamp, long incSum, long decSum) {
045                    setName(name);
046                    setNote(note);
047                    setFirstUsage(firstUsage);
048                    setLastUsage(lastUsage);
049                    setLastReset(lastReset);
050    
051                    setCounter(counter);
052                    setMin(min);
053                    setMax(max);
054                    setMinTimestamp(minTimestamp);
055                    setMaxTimestamp(maxTimestamp);
056                    setIncrementSum(incSum);
057                    setDecrementSum(decSum);
058            }
059    
060            /**
061             * Internal, framework constructor for Simon MBean implementation to initialize all properties
062             * by sample obtained from Simon.
063             *
064             * @param sample sample object obtained from Counter Simon
065             */
066            CounterSample(org.javasimon.CounterSample sample) {
067                    setName(sample.getName());
068                    setNote(sample.getNote());
069                    setFirstUsage(sample.getFirstUsage());
070                    setLastUsage(sample.getLastUsage());
071                    setLastReset(sample.getLastReset());
072    
073                    setCounter(sample.getCounter());
074                    setMin(sample.getMin());
075                    setMax(sample.getMax());
076                    setMinTimestamp(sample.getMinTimestamp());
077                    setMaxTimestamp(sample.getMaxTimestamp());
078                    setIncrementSum(sample.getIncrementSum());
079                    setDecrementSum(sample.getDecrementSum());
080            }
081    
082            /**
083             * Timestamp of the first usage from the sampled Simon as a formatted string.
084             *
085             * @return Simon's first usage timestamp as string
086             */
087            public String getFirstUsageAsString() {
088                    return SimonUtils.presentTimestamp(getFirstUsage());
089            }
090    
091            /**
092             * Timestamp of the first usage from the sampled Simon as a formatted date.
093             *
094             * @return Simon's first usage timestamp as date
095             */
096            public Date getFirstUsageAsDate() {
097                    return new Date(getFirstUsage());
098            }
099    
100            /**
101             * Timestamp of the last usage from the sampled Simon as a formatted string.
102             *
103             * @return Simon's last usage timestamp as string
104             */
105            public String getLastUsageAsString() {
106                    return SimonUtils.presentTimestamp(getLastUsage());
107            }
108    
109            /**
110             * Timestamp of the last usage from the sampled Simon as a date.
111             *
112             * @return Simon's last usage timestamp as date
113             */
114            public Date getLastUsageAsDate() {
115                    return new Date(getLastUsage());
116            }
117    
118            /**
119             * Timestamp of the last reset from the sampled Simon as a formatted string.
120             *
121             * @return Simon's last reset timestamp as string
122             */
123            public String getLastResetAsString() {
124                    return SimonUtils.presentTimestamp(getLastReset());
125            }
126    
127            /**
128             * Timestamp of the last reset from the sampled Simon as a formatted date.
129             *
130             * @return Simon's last reset timestamp as date
131             */
132            public Date getLastResetAsDate() {
133                    return new Date(getLastReset());
134            }
135    
136            /**
137             * Returns ms timestamp when the min value was measured as a formatted string.
138             *
139             * @return ms timestamp of the min value measurement as string
140             */
141            public final String getMinTimestampAsString() {
142                    return SimonUtils.presentTimestamp(getMinTimestamp());
143            }
144    
145            /**
146             * Returns ms timestamp when the min value was measured as a formatted date.
147             *
148             * @return ms timestamp of the min value measurement as date
149             */
150            public final Date getMinTimestampAsDate() {
151                    return new Date(getMinTimestamp());
152            }
153    
154            /**
155             * Returns ms timestamp when the max value was measured as a formatted string.
156             *
157             * @return ms timestamp of the max value measurement as string
158             */
159            public final String getMaxTimestampAsString() {
160                    return SimonUtils.presentTimestamp(getMaxTimestamp());
161            }
162    
163            /**
164             * Returns ms timestamp when the max value was measured as a formatted date.
165             *
166             * @return ms timestamp of the max value measurement as date
167             */
168            public final Date getMaxTimestampAsDate() {
169                    return new Date(getMaxTimestamp());
170            }
171    }
