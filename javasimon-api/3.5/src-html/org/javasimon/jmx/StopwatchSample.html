001    package org.javasimon.jmx;
002    
003    import org.javasimon.utils.SimonUtils;
004    
005    import java.beans.ConstructorProperties;
006    import java.util.Date;
007    
008    /**
009     * Value object for retrieving data from Stopwatch Simon. Basically, it's
010     * {@link org.javasimon.StopwatchSample} with added JMX capabilities to be return as object via
011     * MXBean method.
012     * <p/>
013     * Example:
014     * <pre>
015     * SimonManagerMXBean simon = JMX.newMXBeanProxy(..., new ObjectName("domain:type=Simon"), SimonManagerMXBean.class);
016     * StopwatchSample = simon.getStopwatchSample("simon.stopwatch");
017     * </pre>
018     *
019     * @author Radovan Sninsky
020     * @since 2.0
021     */
022    public final class StopwatchSample extends org.javasimon.StopwatchSample {
023    
024            /**
025             * JMX constructor. Constructor used by JMX client code to initialize all properties of object
026             * from composite data object.
027             *
028             * @param name Simon's name
029             * @param mean mean value (provided optionally)
030             * @param stdDev standard deviation (provided optionally)
031             * @param var variance (provided optionally)
032             * @param varN variance N (provided optionally)
033             * @param firstUsage first usage ms timestamp
034             * @param lastUsage last usage ms timestamp
035             * @param lastReset last reset ms timestamp
036             * @param total sum of all measured times
037             * @param note note (provided optionally)
038             * @param counter count of measures
039             * @param min minimal measured time
040             * @param max maximal measured time
041             * @param minTimestamp ms timestamp when minimal time was measured
042             * @param maxTimestamp ms timestamp when maximal time was measured
043             * @param active count of actual running measures
044             * @param maxActive maximum parallel measures
045             * @param maxActiveTimestamp ms timestamp time when maximum parallel measures happened
046             * @param last last split value in ns
047             */
048            @ConstructorProperties({"name", "mean", "standardDeviation", "variance", "varianceN", "note", "firstUsage", "lastUsage",
049                    "lastReset", "total", "counter", "min", "max", "minTimestamp", "maxTimestamp", "active", "maxActive",
050                    "maxActiveTimestamp", "last"})
051            public StopwatchSample(String name, double mean, double stdDev, double var, double varN, String note, long firstUsage,
052                    long lastUsage, long lastReset, long total, long counter, long min, long max, long minTimestamp,
053                    long maxTimestamp, long active, long maxActive, long maxActiveTimestamp, long last)
054            {
055                    setName(name);
056                    setMean(mean);
057                    setStandardDeviation(stdDev);
058                    setVariance(var);
059                    setVarianceN(varN);
060                    setNote(note);
061                    setFirstUsage(firstUsage);
062                    setLastUsage(lastUsage);
063                    setLastReset(lastReset);
064    
065                    setTotal(total);
066                    setCounter(counter);
067                    setMin(min);
068                    setMax(max);
069                    setMinTimestamp(minTimestamp);
070                    setMaxTimestamp(maxTimestamp);
071                    setActive(active);
072                    setMaxActive(maxActive);
073                    setMaxActiveTimestamp(maxActiveTimestamp);
074                    setLast(last);
075            }
076    
077            /**
078             * Framework constructor for Simon MBean implementation to initialize all properties
079             * by sample obtained from Simon.
080             *
081             * @param sample sample object obtained from Stopwatch Simon
082             */
083            public StopwatchSample(org.javasimon.StopwatchSample sample) {
084                    setName(sample.getName());
085                    setMean(sample.getMean());
086                    setStandardDeviation(sample.getStandardDeviation());
087                    setVariance(sample.getVariance());
088                    setVarianceN(sample.getVarianceN());
089                    setNote(sample.getNote());
090                    setFirstUsage(sample.getFirstUsage());
091                    setLastUsage(sample.getLastUsage());
092                    setLastReset(sample.getLastReset());
093    
094                    setCounter(sample.getCounter());
095                    setTotal(sample.getTotal());
096                    setMin(sample.getMin());
097                    setMax(sample.getMax());
098                    setMinTimestamp(sample.getMinTimestamp());
099                    setMaxTimestamp(sample.getMaxTimestamp());
100                    setActive(sample.getActive());
101                    setMaxActive(sample.getMaxActive());
102                    setMaxActiveTimestamp(sample.getMaxActiveTimestamp());
103                    setLast(sample.getLast());
104            }
105    
106            /**
107             * Returns the value of the last measured split in ns as a formatted string.
108             *
109             * @return last measured split in ns as string
110             */
111            public final String getLastAsString() {
112                    return SimonUtils.presentNanoTime(getLast());
113            }
114    
115            /**
116             * Returns the total sum of all split times in nanoseconds as a formatted string.
117             *
118             * @return total time of the stopwatch in nanoseconds as string
119             */
120            public final String getTotalAsString() {
121                    return SimonUtils.presentNanoTime(getTotal());
122            }
123    
124            /**
125             * Returns minimal time split value in nanoseconds as a formatted string.
126             *
127             * @return minimal time split in nanoseconds as string
128             */
129            public final String getMinAsString() {
130                    return SimonUtils.presentNanoTime(getMin());
131            }
132    
133            /**
134             * Returns maximal time split value in nanoseconds as a formatted string.
135             *
136             * @return maximal time split in nanoseconds as string
137             */
138            public final String getMaxAsString() {
139                    return SimonUtils.presentNanoTime(getMax());
140            }
141    
142            /**
143             * Returns ms timestamp when the min value was measured as a formatted string.
144             *
145             * @return ms timestamp of the min value measurement as string
146             */
147            public final String getMinTimestampAsString() {
148                    return SimonUtils.presentTimestamp(getMinTimestamp());
149            }
150    
151            /**
152             * Returns ms timestamp when the min value was measured as a formatted date.
153             *
154             * @return ms timestamp of the min value measurement as date
155             */
156            public final Date getMinTimestampAsDate() {
157                    return new Date(getMinTimestamp());
158            }
159    
160            /**
161             * Returns ms timestamp when the max value was measured as a formatted string.
162             *
163             * @return ms timestamp of the max value measurement as string
164             */
165            public final String getMaxTimestampAsString() {
166                    return SimonUtils.presentTimestamp(getMaxTimestamp());
167            }
168    
169            /**
170             * Returns ms timestamp when the max value was measured as a formatted date.
171             *
172             * @return ms timestamp of the max value measurement as date
173             */
174            public final Date getMaxTimestampAsDate() {
175                    return new Date(getMaxTimestamp());
176            }
177    
178            /**
179             * Returns ms timestamp when the last peek of the active split count occurred as a formatted string.
180             *
181             * @return ms timestamp of the last peek of the active split count as string
182             */
183            public final String getMaxActiveTimestampAsString() {
184                    return SimonUtils.presentTimestamp(getMaxActiveTimestamp());
185            }
186    
187            /**
188             * Returns ms timestamp when the last peek of the active split count occurred as a formatted date.
189             *
190             * @return ms timestamp of the last peek of the active split count as date
191             */
192            public final Date getMaxActiveTimestampAsDate() {
193                    return new Date(getMaxActiveTimestamp());
194            }
195    
196            /**
197             * Returns mean value (average) of all measured values as a formatted string (ns).
198             *
199             * @return mean value as string
200             */
201            public final String getMeanAsString() {
202                    return SimonUtils.presentNanoTime((long) getMean());
203            }
204    
205            /**
206             * Timestamp of the first usage from the sampled Simon as a formatted string.
207             *
208             * @return Simon's first usage timestamp as string
209             */
210            public String getFirstUsageAsString() {
211                    return SimonUtils.presentTimestamp(getFirstUsage());
212            }
213    
214            /**
215             * Timestamp of the first usage from the sampled Simon as a formatted date.
216             *
217             * @return Simon's first usage timestamp as date
218             */
219            public Date getFirstUsageAsDate() {
220                    return new Date(getFirstUsage());
221            }
222    
223            /**
224             * Timestamp of the last usage from the sampled Simon as a formatted string.
225             *
226             * @return Simon's last usage timestamp as string
227             */
228            public String getLastUsageAsString() {
229                    return SimonUtils.presentTimestamp(getLastUsage());
230            }
231    
232            /**
233             * Timestamp of the last usage from the sampled Simon as a formatted date.
234             *
235             * @return Simon's last usage timestamp as string
236             */
237            public Date getLastUsageAsDate() {
238                    return new Date(getLastUsage());
239            }
240    
241            /**
242             * Timestamp of the last reset from the sampled Simon as a formatted string.
243             *
244             * @return Simon's last reset timestamp as string
245             */
246            public String getLastResetAsString() {
247                    return SimonUtils.presentTimestamp(getLastReset());
248            }
249    
250            /**
251             * Timestamp of the last reset from the sampled Simon as a formatted date.
252             *
253             * @return Simon's last reset timestamp as date
254             */
255            public Date getLastResetAsDate() {
256                    return new Date(getLastReset());
257            }
