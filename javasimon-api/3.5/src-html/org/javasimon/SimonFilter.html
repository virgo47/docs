001    package org.javasimon;
002    
003    /**
004     * Generic filter useful whenever Simons are filtered for some operation.
005     *
006     * @author <a href="mailto:ivan.mushketyk@gmail.com">Ivan Mushketyk</a>
007     */
008    public interface SimonFilter {
009    
010            /**
011             * Checks whether current Simon should be used/considered.
012             *
013             * @param simon Simon to check
014             * @return true if current Simon should be used, false otherwise
015             */
016            boolean accept(Simon simon);
017    }
