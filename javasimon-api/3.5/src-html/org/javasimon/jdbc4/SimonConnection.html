025    /**
026     * Class implements Simon JDBC proxy connection.
027     * <p>
028     * Every method of this connection is implemented as call of real connection method.
029     * Several methods have added work with Simons (starting, stoping, etc.) for monitoring
030     * purposes.
031     * </p>
032     * <p>
033     * From all statement-return-methods (<code>createStatement(*)</code>,
034     * <code>prepareStatement(*)</code>, <code>prepareCall(*)</code>) connection returns own
035     * implementation of statement classes. Those classes are also proxies and provides
036     * additional Simons for monitoring features of JDBC driver.
037     * </p>
038     * Monitoring connection ensure following Simons:
039     * <ul>
040     * <li>lifespan (<code>org.javasimon.jdbc4.conn</code>, stopwatch) - measure connection life and count</li>
041     * <li>commits (<code>org.javasimon.jdbc4.conn.commits</code>, counter) - measure executed commits of all connections</li>
042     * <li>rollbacks (<code>org.javasimon.jdbc4.conn.rollbacks</code>, counter) - measure executed rollbacks of all connections</li>
043     * </ul>
044     *
045     * @author Radovan Sninsky
046     * @see java.sql.Connection
047     * @since 2.4
048     */
049    public final class SimonConnection implements Connection {
050            private final Connection conn;
051            private final WrapperSupport<Connection> wrapperSupport;
052            private String suffix;
053    
054            private Split life;
055            private Counter commits;
056            private Counter rollbacks;
057    
058            /**
059             * Class constructor, initializes Simons (lifespan, active, commits
060             * and rollbacks) related to the DB connection.
061             *
062             * @param conn real DB connection
063             * @param prefix hierarchy prefix for connection Simons
064             */
065            public SimonConnection(Connection conn, String prefix) {
066                    this.conn = conn;
067                    this.wrapperSupport = new WrapperSupport<Connection>(this.conn, Connection.class);
068                    this.suffix = prefix;
069    
070                    commits = SimonManager.getCounter(prefix + ".conn.commits");
071                    rollbacks = SimonManager.getCounter(prefix + ".conn.rollbacks");
072                    life = SimonManager.getStopwatch(prefix + ".conn").start();
073            }
074    
075            /**
076             * Closes the real connection, stops lifespan Simon and decreases the active Simon.
077             *
078             * @throws java.sql.SQLException if real operation fails
079             */
080            @Override
081            public void close() throws SQLException {
082                    conn.close();
083    
084                    life.stop();
085            }
086    
087            /**
088             * Commits the real connection and increases the commits Simon.
089             *
090             * @throws java.sql.SQLException if real commit fails
091             */
092            @Override
093            public void commit() throws SQLException {
094                    conn.commit();
095    
096                    commits.increase();
097            }
098    
099            /**
100             * Rollback the real connection and increase the rollbacks Simon.
101             *
102             * @throws java.sql.SQLException if real operation fails
103             */
104            @Override
105            public void rollback() throws SQLException {
106                    conn.rollback();
107    
108                    rollbacks.increase();
109            }
110    
111            /**
112             * Rollbacks the real connection and increases the rollbacks Simon.
113             *
114             * @param savepoint the <code>Savepoint</code> object to roll back to
115             * @throws java.sql.SQLException if real operation fails
116             */
117            @Override
118            public void rollback(Savepoint savepoint) throws SQLException {
119                    conn.rollback(savepoint);
120    
121                    rollbacks.increase();
122            }
