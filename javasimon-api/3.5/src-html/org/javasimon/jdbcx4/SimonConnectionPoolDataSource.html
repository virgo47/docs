001    package org.javasimon.jdbcx4;
002    
003    import java.sql.SQLException;
004    import javax.sql.ConnectionPoolDataSource;
005    import javax.sql.PooledConnection;
006    
007    /**
008     * Wrapper class for real ConnectionPoolDataSource implementation, produces pooled
009     * {@link javax.sql.PooledConnection} object.
010     * <p/>
011     * See the {@link org.javasimon.jdbcx4.SimonDataSource} for more information.
012     *
013     * @author Radovan Sninsky
014     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
015     * @since 2.4
016     */
017    public final class SimonConnectionPoolDataSource extends AbstractSimonDataSource implements ConnectionPoolDataSource {
018            private ConnectionPoolDataSource ds;
019    
020            private ConnectionPoolDataSource datasource() throws SQLException {
021                    if (ds == null) {
022                            ds = createDataSource(ConnectionPoolDataSource.class);
023                    }
024                    return ds;
025            }
026    
027            @Override
028            public PooledConnection getPooledConnection() throws SQLException {
029                    return new SimonPooledConnection(datasource().getPooledConnection(), getPrefix());
030            }
031    
032            @Override
033            public PooledConnection getPooledConnection(String user, String password) throws SQLException {
034                    return new SimonPooledConnection(datasource().getPooledConnection(user, password), getPrefix());
035            }
036    
037            @Override
038            protected String doGetRealDataSourceClassName() {
039                    return configuration.getRealConnectionPoolDataSourceName();
040            }
041    }
