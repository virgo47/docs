001package org.javasimon.utils.bean;
002
003/**
004 * Converter from String to Float.
005 *
006 * @author <a href="mailto:ivan.mushketyk@gmail.com">Ivan Mushketyk</a>
007 */
008public class ToFloatConverter implements Converter {
009
010        @Override
011        public Float convert(Class<?> targetClass, String strVal) throws ConvertException {
012                if (strVal == null) {
013                        return null;
014                }
015
016                try {
017                        return Float.parseFloat(strVal);
018                } catch (NumberFormatException ex) {
019                        throw new ConvertException(String.format("Failed to parse '%s' to Float", strVal));
020                }
021        }
022}
