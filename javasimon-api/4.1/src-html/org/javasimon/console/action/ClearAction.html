001package org.javasimon.console.action;
002
003import org.javasimon.console.Action;
004import org.javasimon.console.ActionContext;
005import org.javasimon.console.ActionException;
006
007import java.io.IOException;
008
009import javax.servlet.ServletException;
010
011/**
012 * Action to clear the Simon manager and remove all Simons.
013 *
014 * @author gquintana
015 */
016public class ClearAction extends Action {
017
018        /** URI for clear action. */
019        public static final String PATH = "/data/clear";
020
021        public ClearAction(ActionContext context) {
022                super(context);
023        }
024
025        @Override
026        public void execute() throws ServletException, IOException, ActionException {
027                getContext().getManager().clear();
028        }
029}
